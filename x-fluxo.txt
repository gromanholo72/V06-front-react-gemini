// |   |-- 📄🗺️ App.jsx (A Planta da Casa - Onde o Gerente/Cérebro trabalha)

// |   +-- 📄 interno.jsx (A Sala VIP - Rota Restrita)
// |
// |-- // 3 - Os Cômodos (Componentes/Páginas que o usuário vê)
// |   |
// |   |-- 🚪 Cadastro (Recepção - Caminho: /)
// |   |
// |   |-- 🚪 Rota "/cadastro" -> Leva ao cômodo Cadastrar.jsx
// |   |
// |   |-- 🚪 Rota "/login"    -> Leva ao cômodo Logar.jsx

// |
// +-- // 4 - Entidades (Software/Lógica)
// |    |
// |    |-- 🏗️ Node-Client (O escritorio no 🏢 Servidor/VPS e o arquiteto do node)
// |    |    |
// |    |    |-- É o ambiente que sustenta o canteiro de obras no seu computador.
// |    |    |-- Fornece a energia e as ferramentas para o Montador (Vite) trabalhar.
// |    |    |-- Sem esse escritório local, o Montador não consegue organizar os materiais.
// |    |    |
// |    |    +-- 📦 node_modules (O Estoque): Onde ficam guardadas todas as peças e ferramentas.
// |    |
// |    |-- 🔩 Vite (Montador/Desmontador da Casa - Ajudante do Arquiteto/Mestre) e entrega no navegador
// |    |   |
// |    |   |-- Preparou os materiais e leu os projetos (Imports).
// |    |   |-- Vive dentro do ecossistema do 🏗️ Node.
// |    |   |-- Usa o motor do Node para organizar os materiais rapidamente.
// |    |   |-- Ferramenta que organiza e monta os materiais da casa.
// |    |   |-- Garante que o Gerente (React) veja as mudanças na hora.
// |    |   |
// |    |   |-- 👷 createRoot: A ferramenta do Mestre de Obras para iniciar a fundação.
// |    |   |
// |    |   +-- 🧠 StrictMode: O Fiscal de Obras que vigia erros no código.
// |    |
// |    +-- 🧠 React (Engenheiro da Obra - Design e Inteligência) - A autoridade técnica no canteiro
// |        |
// |        |-- Toma todas as decisões baseadas no que o usuário faz.
// |        |-- Gerencia o estado (se a luz está acesa ou apagada).
// |        |-- Decide quando é hora de usar o rádio ou abrir a porta.
// |        |
// |        |-- 🧰 Hooks (Ferramentas de Trabalho) - O kit completo de recursos do Engenheiro
// |        |   |
// |        |   |-- 🗂️ useState (Gavetas de Memória) - Guarda variáveis e estados
// |        |   |
// |        |   |-- 📅 useEffect (A Agenda de Monitoramento) - vigilância e reação
// |        |   |
// |        |   |   |-- 🛠️ AÇÃO: Ele verifica se as ferramentas (Hooks) estão no lugar.
// |        |   |
// |        |   |   +-- 📶 AÇÃO: Ele olha para a antena (Wi-Fi) para ver se tem sinal.
// |        |   |
// |        |   +-- 🚕 useNavigate (O Motorista): O Hook que contrata o piloto para a viagem.
// |        |
// |        |-- 🛣️ React Router (Corredores Virtuais / react-router-dom) - Infraestrutura de Trânsito
// |        |   |
// |        |   |-- 🛣 BrowserRouter (O Asfalto): O ambiente que permite a navegação.
// |        |   |
// |        |   |-- 🚩 Routes/Route (Placas de Sinalização): Define os caminhos.
// |        |   |   |-- 🚪 Rota "/cadastro" -> Leva ao cômodo Cadastrar.jsx
// |        |   |   |-- 🚪 Rota "/login"    -> Leva ao cômodo Logar.jsx
// |        |   |   +-- 🚪 Rota "/painel"   -> Leva ao cômodo Painel.jsx
// |        |   |
// |        |   |--- 🚕 Comando navigate('/rota'): O comando de partida executado pelo Motorista.
// |        |   |
// |        |   +-- 🔐 Guarda (Middleware): Proteção que impede visitas sem crachá.
// |        |
// |        |-- ⚙️ Estrutura Híbrida de Navegação:
// |        |   |
// |        |   |-- 🖼️ Estado (setTelaAtiva): Troca divisórias internas para o Visitante.
// |        |   |
// |        |   +-- 🛣️ Rota Virtual (/logado): Ganha um endereço próprio após o cadastro.
// |        |
// |        |-- [ 📡📻 Socket-Client (Rádio Portátil) ]
// |        |   |
// |        |   |-- 📞 Socket.io ( Mensageiro)
// |        |   |    |-- É a LINGUAGEM ou o "Aparelho" que os dois usam.
// |        |   |    +-- Sem ele, a Antena e o Rádio falariam línguas diferentes.
// |        |   |
// |        |   |-- 🛰️🟢 socket.on('connect') (O Ponto de Contato estabelecido)
// |        |   |    - EVENTO ESPECIAL de fábrica.
// |        |   |    - Ele só acontece uma vez (no momento em que a conexão é feita)
// |        |   |    - Você não escolhe o nome dele; ele já vem de fábrica

// |        |   |
// |        |   |-- ✈️ socket.emit (Fala com o Prédio)

// |        |   |
// |        |   +-- 📬 socket.on   (Escuta o Prédio)

// 🏗️ Node-Server (Empresa de Engenharia dentro do 🏢 Servidor/VPS)
// |    | Descrição:
// |    | ---- Back-end
// |    | ---- inteligência central
// |    | ---- opera o motor oculto no porão do prédio que governa toda a obra.
// |    |
// |    |-- 📐 Escritorio do Node-Server (Localização)  - dentro do Prédio
// |    +-- 🌈 arquiteto do Node-Server (Mente do Projeto) - nunca sai do Prédio
// |
// |-- // 1 - Infraestrutura de Base (O Terreno Técnico)
// |   |
// |   |-- ⚡ Runtime (Node.js) - O chão de fábrica onde o motor gira.
// |   |-- 🌐 Endereço (IP/DNS) - As coordenadas para as casas acharem o prédio.
// |   +-- 🔓 Porta (Ex: 3001) - O portão específico de entrada do fluxo.
// |
// |-- // 2 - Documentos (Arquivos)
// |   |
// |   |-- 📄 server.js (O Projeto do Prédio - Onde o Arquiteto escreve as regras)
// |   |-- 📄 package.json (A Lista de Materiais e Ferramentas do Prédio)
// |   +-- 📄 .env (O Cofre de Segredos - Senhas e Chaves)
// |
// |-- // 3 - Entidades (Software/Lógica) 
// |   | 
// |   |-- operação em tempo real
// |   |-- É o motor girando, 
// |   |-- a 📡🗼 Antena Central ligada 
// |   |-- o🛂 Porteiro (Express) trabalhando.
// |   |-- O motor que faz o Prédio existir e funcionar.
// |   |-- É quem contrata o Mensageiro e o Segurança (Express).
// |   |-- Sustenta e opera:
// |   |
// |   |-- 📡🗼 Socket-Server (Antena Central) 
// |   |    |
// |   |    |-- 👂 "io" (Grande Ouvinte) - O Sistema da Antena ligado 24h.
// |   |    |    |
// |   |    |    |-- 🔢 io.engine.clientsCount (Mostrador de Ocupação no painel do "io").
// |   |    |    |
// |   |    |    |-- 🚀 io.emit (Grito Geral): O "io" fala com todas as casas ao mesmo tempo.
// |   |    |    |
// |   |    |    +-- 🤝 io.on('connection') (Aperto de Mão): O "io" detecta um novo rádio.
// |   |    |        |   // o prédio é um receptor central
// |   |    |        |   // Ele não "se conecta"
// |   |    |        |   // ele recebe conexões
// |   |    |        |   // O evento diz: "Alguém se conectou a mim!".
// |   |    |        |
// |   |    |        |-- 🆔 socket.id (Identidade única gerada para aquela 🏠 casa).
// |   |    |        |
// |   |    |        |-- ✈️ socket.emit (Mensagem Direta): Prédio fala com ESSA casa.
// |   |    |        +-- 📬 socket.on (Escuta): Prédio ouve o que ESSA casa diz.
// |   |    |
// |   |--  [ 📄 Express/Rotas (O Livro de Regras) ]
// |   |    |-- É o porteiro do prédio: diz quem pode ou não entrar.
// |   |    +-- Define os caminhos (corredores) para chegar aos dados.
// |   |
// |   +-- [ 📦 Firebase (O Almoxarifado - Onde tudo é guardado) ]
// |        |-- Banco de dados onde guardamos as plantas e registros.
// |        +-- Fica fora do prédio, mas o Arquiteto tem a chave mestra.
// |
// +-- [ 🗂️ Gaveta da Memória (RAM - Dados Temporários) ]
//      |-- Onde o Arquiteto anota coisas rápidas que não precisam ser guardadas para sempre.
//      +-- Se o prédio cair, essa gaveta esvazia.
